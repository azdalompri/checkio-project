def check_connection(network, first, second):
    d = dict()
    for i in network:
        p1, p2 = i.split('-')
        d.setdefault(p1, []).append(p2)
        d.setdefault(p2, []).append(p1)
    opn = [first]
    cls = []
    while len(opn):
        now = opn.pop()
        cls.append(now)
        if now == second:
            return True
        for next in d.setdefault(now, []):
            if next not in cls and next not in opn:
                opn.append(next)
    return False
